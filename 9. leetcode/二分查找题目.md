# 搜索插入位置

```
int num = 3;
cout << num/2 << endl;    // 1
```

```C++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int middle, left = 0, right = nums.size() -1;
        while(left <= right)
        {
            middle = (right - left) /2 + left;
            if(nums[middle] == target)  return middle;
            if(target < nums[middle])   right = middle -1;
            else    left = middle + 1;
        }
        return left;
    }
};
```
